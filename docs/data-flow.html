export function useAresSearch() {
  const queryClient = useQueryClient();

  return useMutation({
    // This is the actual API call (step 4)
    mutationFn: (params) => aresEndpoints.search(params),

    onSuccess: (data, variables) => {
      // Cache the full search result
      queryClient.setQueryData(aresKeys.search(variables), data);

      // Pre-cache each company for instant detail page loads
      data.economicSubjects.forEach((subject) => {
        queryClient.setQueryData(aresKeys.detail(subject.icoId), subject);
      });
    },
  });
}
Why this matters: When you click a result card to view company details, the data is already cached — no extra API call needed. This is the frontend cache (TanStack Query in browser memory). There's also a backend cache (Redis) which we'll see later.

class AresService:
    def __init__(self):
        self.client = ares_client
        self.cache = CacheService(prefix="ares", default_ttl=900)
        self.outbound_throttle = GlobalOutboundThrottle(
            key="ares", max_requests=12, window=60  # 12 req/min
        )

    def search(self, params: dict) -> dict:
        # 1. Translate English field names to Czech for ARES API
        request_body = to_search_request(params)

        # 2. Check Redis cache (SHA256 hash of params)
        cache_hash = self.cache.hash_params(request_body)
        cached = self.cache.get("search", cache_hash)
        if cached is not None:
            return cached  # ← CACHE HIT! Skip steps 11-13

        # 3. Rate limiting: max 12 outbound requests per minute
        if not self.outbound_throttle.allow():
            raise ExternalAPIError("Rate limit reached", status_code=429)

        # 4. Call the Czech government ARES API
        raw = self.client.search(request_body)

        # 5. Parse Czech response to English
        result = parse_search_result(raw)

        # 6. Cache the result in Redis (15 min for search, 1 hr for detail)
        self.cache.set(result, "search", cache_hash, ttl=900)
        for subject in result.get("economicSubjects", []):
            ico_id = subject.get("icoId")
            if ico_id:
                self.cache.set(subject, "detail", ico_id, ttl=3600)

        return result
This is the most important backend file. It orchestrates: translation → cache check → rate limit → external API call → response parsing → cache storage.

▼
backend/core/services/cache.py 
class CacheService:
    def __init__(self, prefix="ares", default_ttl=900):
        self.prefix = prefix

    def _make_key(self, *parts):
        # "ares" + "search" + "a1b2c3d4" → "ares:search:a1b2c3d4"
        return f"{self.prefix}:{':'.join(parts)}"

    def hash_params(self, params: dict) -> str:
        # Deterministic SHA256 hash of search params
        serialized = json.dumps(params, sort_keys=True, default=str)
        return hashlib.sha256(serialized.encode()).hexdigest()[:16]

    def get(self, *key_parts):
        return cache.get(self._make_key(*key_parts))

    def set(self, value, *key_parts, ttl=None):
        cache.set(self._make_key(*key_parts), value, ttl or self.default_ttl)
Cache keys explained:
Search: ares:search:a1b2c3d4e5f6g7h8 (hash of query params) — TTL 15 min
Detail: ares:detail:12345678 (the ICO itself) — TTL 1 hour

def parse_search_result(response: dict) -> dict:
    return {
        "totalCount": response["pocetCelkem"],        # Czech → English
        "economicSubjects": [
            parse_economic_subject(s)
            for s in response.get("ekonomickeSubjekty", [])
        ],
    }

def parse_economic_subject(subject: dict) -> dict:
    record = {
        "ico":            subject["ico"],
        "businessName":   subject["obchodniJmeno"],    # Czech → English
        "headquarters":   parse_headquarters(subject.get("sidlo")),
        "legalForm":      subject.get("pravniForma"),
        "foundationDate": subject.get("datumVzniku"),
        "vatId":          subject.get("dic"),
    }
    return { "icoId": subject["ico"], "records": [record] }

The Justice module has three data pipelines: (A) Batch sync from the Czech open data portal into PostgreSQL, (B) Entity lookup API serving stored data to the frontend, and (C) Live document scraping from the Czech court registry (Sbirka listin).

    def list_datasets(self) -> list[str]:
        # GET /api/3/action/package_list
        resp = self.session.get(f"{CKAN_API}/package_list")
        return resp.json()["result"]  # ["sro-actual-praha-2026", ...]

    def get_dataset(self, dataset_id: str) -> dict:
        # GET /api/3/action/package_show?id=sro-actual-praha-2026
        resp = self.session.get(f"{CKAN_API}/package_show", params={"id": dataset_id})
        return resp.json()["result"]  # { "resources": [{"url": "...xml.gz"}] }

    def get_file_size(self, filename: str) -> int | None:
        # HEAD request for Content-Length (change detection)
        resp = self.session.head(f"{FILE_API}/{filename}")
        return int(resp.headers.get("Content-Length"))

    def download_file_stream(self, filename: str) -> Iterator[bytes]:
        # GET /api/file/{filename} with stream=True
        resp = self.session.get(f"{FILE_API}/{filename}", stream=True)
        return resp.iter_content(chunk_size=65536)
        # Returns byte chunks — never loads full file into memory
Streaming is critical: These XML files can be hundreds of MB compressed. The stream=True parameter tells requests to yield chunks instead of loading the entire response into memory. Each 64KB chunk flows directly into the gzip decompressor and XML parser.

Backend — Streaming XML Parser
A4
backend/justice/parsers/xml_parser.py
Streaming XML parser using lxml.etree.iterparse. Decompresses gzip, processes one <Subjekt> at a time, and frees memory after each record.
▼
backend/justice/parsers/xml_parser.py 
def parse_xml_stream(data_stream: Iterator[bytes]) -> Iterator[dict]:
    # 1. Buffer the gzip stream (needed for GzipFile)
    raw_buffer = BytesIO()
    for chunk in data_stream:
        raw_buffer.write(chunk)
    raw_buffer.seek(0)

    # 2. Decompress and iterparse one <Subjekt> at a time
    with gzip.GzipFile(fileobj=raw_buffer) as gz_file:
        context = etree.iterparse(gz_file, events=("end",), tag="Subjekt")
        for _event, elem in context:
            yield _parse_subjekt(elem)
            # Free memory: clear element and remove siblings
            elem.clear()
            while elem.getprevious() is not None:
                del elem.getparent()[0]

def _parse_subjekt(elem) -> dict:
    return {
        "name": _text(elem, "nazev"),         # Company name
        "ico": _text(elem, "ico"),             # ID number
        "registration_date": _text(elem, "zapisDatum"),
        "deletion_date": _text(elem, "vymazDatum"),
        "facts": [_parse_udaj(u) for u in elem.findall("udaje/Udaj")],
    }
Memory management: The iterparse pattern processes XML elements one at a time. After yielding each <Subjekt>, the parser calls elem.clear() and removes preceding siblings from the tree. This keeps memory usage constant regardless of file size — essential for files with tens of thousands of entities.

    # 3. Bulk create level by level (resolves parent FKs)
    for depth in range(max_depth + 1):
        created = EntityFact.objects.bulk_create(objs_at_depth)
        # Map collect_index → real PK for parent references

Level-by-level bulk creation: Facts form a tree (parent → sub_facts). This turns N individual INSERTs into ~3 bulk INSERTs.

class JusticeService:
    def __init__(self):
        self.cache = CacheService(prefix="justice", default_ttl=900)

    def get_entity_by_ico(self, ico: str) -> dict:
        # 1. Normalize ICO (zero-pad to 8 digits)
        normalized = ico.zfill(8)

        # 2. Check Redis cache
        cached = self.cache.get("entity", normalized)
        if cached is not None:
            return cached  # CACHE HIT — skip DB query

        # 3. Query PostgreSQL with eager loading
        entity = Entity.objects.filter(ico=normalized).first()
        facts = EntityFact.objects.filter(entity=entity) \
            .select_related("person") \
            .prefetch_related("addresses", "sub_facts") \
            .order_by("registration_date")

        # 4. Transform model instances to camelCase API dicts
        result = parse_entity_detail(entity, list(facts))

        # 5. Cache for 1 hour
        self.cache.set(result, "entity", normalized, ttl=3600)
        return result
Key difference from ARES: ARES calls an external government API on cache miss. Justice reads from local PostgreSQL — the data was already ingested by Pipeline A. This makes Justice queries much faster and independent of government API availability.

Cache keys: justice:entity:12345678 (1 hr), justice:search:{hash} (15 min), justice:datasets (1 hr), justice:documents:12345678 (1 hr).

# EPO format XML structure:
# <UcetniZaverka>
#   <VetaD>    — metadata (period, currency, format)
#   <VetaUA>   — Rozvaha AKTIVA (balance sheet assets)
#   <VetaUD>   — Rozvaha PASIVA (liabilities/equity)
#   <VetaUB>   — VZZ P&L (income statement)

# Row numbers are legally standardized (Vyhláška 500/2002 Sb.)
VZZ_ROWS = {
    1: "I. Tržby z prodeje výrobkú a služeb",
    2: "II. Tržby za prodej zboží",
    3: "A. Výkonová spotřeba",
    ...  # 56 standardized line items
}

def parse_financial_xml(xml_bytes: bytes) -> dict:
    root = etree.fromstring(xml_bytes)
    return {
        "metadata": _parse_metadata(root),  # period, format, currency
        "aktiva": _parse_section(root, "VetaUA", AKTIVA_ROWS),
        "pasiva": _parse_section(root, "VetaUD", PASIVA_ROWS),
        "vzz": _parse_section(root, "VetaUB", VZZ_ROWS),
    }
What this enables: Instead of showing users raw PDF files, the financial data is parsed into structured JSON with named line items. class DocumentProxyView(APIView):
    def get(self, request, download_id):
        client = justice_sbirka_client

        # Download from or.justice.cz using server-side session
        content, content_type, filename = client.download_file(download_id)

        # Forward as inline response (opens in browser)
        response = HttpResponse(content, content_type=content_type)
        response["Content-Disposition"] = f'inline; filename="{filename}"'
        return response

# Browser: GET /api/v1/justice/documents/{uuid}/

# Django:  GET or.justice.cz/ias/content/download?id={uuid}
#          (with server-side session cookies)
#   ↓
# Browser: receives PDF/XML as if it came from our domain
CORS workaround: The browser can't directly download from or.justice.cz because of cross-origin restrictions and required session cookies. The proxy makes our Django server act as a middleman — it downloads the file server-side (where cookies work) and re-serves it to the browser from our own domain.

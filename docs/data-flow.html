TanStack Query mutation. Calls the API endpoint, then caches the results on success.
▼
src/lib/ares/ares.mutations.ts 
export function useAresSearch() {
  const queryClient = useQueryClient();

  return useMutation({
    // This is the actual API call (step 4)
    mutationFn: (params) => aresEndpoints.search(params),

    onSuccess: (data, variables) => {
      // Cache the full search result
      queryClient.setQueryData(aresKeys.search(variables), data);

      // Pre-cache each company for instant detail page loads
      data.economicSubjects.forEach((subject) => {
        queryClient.setQueryData(aresKeys.detail(subject.icoId), subject);
      });
    },
  });
}
Why this matters: When you click a result card to view company details, the data is already cached — no extra API call needed. This is the frontend cache (TanStack Query in browser memory). There's also a backend cache (Redis) which we'll see later.

The view is the glue layer — it validates input, delegates to the service, and formats the output. Only 4 lines of actual code.

backend/ares/views.py 
class AresSearchView(APIView):
    def post(self, request):
        # 1. Validate incoming JSON (step 9)
        serializer = SearchRequestSerializer(data=request.data)
        serializer.is_valid(raise_exception=True)

        # 2. Call the service layer (step 10)
        service = AresService()
        result = service.search(serializer.validated_data)

        # 3. Format the result back to JSON for the frontend
        return Response(SearchResultSerializer(result).data)
Thin controller pattern: The view doesn't contain any business logic. It follows a strict 3-step pattern: validate input → delegate to service → format output. This means the same AresService.search() can be called from a CLI command, a background job, or a test — without needing HTTP.

backend/ares/services.py
The brain of the operation. Translates params to Czech, checks cache, checks rate limit, calls ARES API, parses response, caches result.
▼
backend/ares/services.py 
class AresService:
    def __init__(self):
        self.client = ares_client
        self.cache = CacheService(prefix="ares", default_ttl=900)
        self.outbound_throttle = GlobalOutboundThrottle(
            key="ares", max_requests=12, window=60  # 12 req/min
        )

    def search(self, params: dict) -> dict:
        # 1. Translate English field names to Czech for ARES API
        request_body = to_search_request(params)
        #    { "ico": ["12345678"] } → { "ico": ["12345678"] }
        #    { "businessName": "X" } → { "obchodniJmeno": "X" }

        # 2. Check Redis cache (SHA256 hash of params)
        cache_hash = self.cache.hash_params(request_body)
        cached = self.cache.get("search", cache_hash)
        if cached is not None:
            return cached  # ← CACHE HIT! Skip steps 11-13

        # 3. Rate limiting: max 12 outbound requests per minute
        if not self.outbound_throttle.allow():
            raise ExternalAPIError("Rate limit reached", status_code=429)

        # 4. Call the Czech government ARES API
        raw = self.client.search(request_body)

        # 5. Parse Czech response to English
        result = parse_search_result(raw)

        # 6. Cache the result in Redis (15 min for search, 1 hr for detail)
        self.cache.set(result, "search", cache_hash, ttl=900)
        for subject in result.get("economicSubjects", []):
            ico_id = subject.get("icoId")
            if ico_id:
                self.cache.set(subject, "detail", ico_id, ttl=3600)

        return result
This is the most important backend file. It orchestrates: translation → cache check → rate limit → external API call → response parsing → cache storage. The to_search_request() function maps English field names to Czech ones that the government API expects.

Cache — Redis
11
backend/core/services/cache.py
Redis cache wrapper. Builds namespaced keys like ares:search:a1b2c3d4 and stores JSON data with TTL.
▼
backend/core/services/cache.py 
class CacheService:
    def __init__(self, prefix="ares", default_ttl=900):
        self.prefix = prefix

    def _make_key(self, *parts):
        # "ares" + "search" + "a1b2c3d4" → "ares:search:a1b2c3d4"
        return f"{self.prefix}:{':'.join(parts)}"

    def hash_params(self, params: dict) -> str:
        # Deterministic SHA256 hash of search params
        serialized = json.dumps(params, sort_keys=True, default=str)
        return hashlib.sha256(serialized.encode()).hexdigest()[:16]

    def get(self, *key_parts):
        return cache.get(self._make_key(*key_parts))

    def set(self, value, *key_parts, ttl=None):
        cache.set(self._make_key(*key_parts), value, ttl or self.default_ttl)
Cache keys explained:
Search: ares:search:a1b2c3d4e5f6g7h8 (hash of query params) — TTL 15 min
Detail: ares:detail:12345678 (the ICO itself) — TTL 1 hour

Redis runs in Docker with max 256MB RAM and LRU eviction (least recently used entries get deleted first when memory is full).

ARES_BASE_URL = "https://ares.gov.cz/ekonomicke-subjekty-v-be/rest/ekonomicke-subjekty"

class AresClient:
    def __init__(self):
        self.session = requests.Session()  # connection pooling

    def search(self, request_body: dict) -> dict:
        resp = self.session.post(
            f"{self.base_url}/vyhledat",  # Czech for "search"
            json=request_body,    # { "ico": ["12345678"] }
            timeout=15,
        )
        resp.raise_for_status()
        return resp.json()        # Raw Czech response

def parse_search_result(response: dict) -> dict:
    return {
        "totalCount": response["pocetCelkem"],        # Czech → English
        "economicSubjects": [
            parse_economic_subject(s)
            for s in response.get("ekonomickeSubjekty", [])
        ],
    }

def parse_economic_subject(subject: dict) -> dict:
    record = {
        "ico":            subject["ico"],
        "businessName":   subject["obchodniJmeno"],    # Czech → English
        "headquarters":   parse_headquarters(subject.get("sidlo")),
        "legalForm":      subject.get("pravniForma"),
        "foundationDate": subject.get("datumVzniku"),
        "vatId":          subject.get("dic"),
        // ... more fields ...
    }
    return { "icoId": subject["ico"], "records": [record] }

# Address fields also get translated:
#   "nazevObce"  → "municipalityName"
#   "nazevUlice" → "streetName"
#   "psc"        → "postalCode"
Why this exists: The Czech government API returns everything in Czech (obchodniJmeno, sidlo, pocetCelkem). This parser converts every field name to English so the frontend works with clean, readable TypeScript types.

Two caches at play:
1. Backend (Redis): Prevents re-calling the government API. TTL: 15 min for search, 1 hr for detail.
2. Frontend (TanStack Query): Prevents re-calling your Django backend. Stale time: 1 min. Cache time: 5 min.

So the second time you search the same ICO within a minute, it doesn't even leave the browser!

export function useCompanySearch(params: CompanySearchParams, options?: SearchOptions) {
  return useQuery({
    queryKey: companyKeys.search(params),
    // Key: ["company", "search", { name: "Tesla", status: "active", ... }]

    queryFn: () => companyEndpoints.search(params),
    // Calls the API function (step 5)

    ...options,  // { enabled: hasSearchCriteria }
  });
}
src/lib/company/company.keys.ts 
export const companyKeys = {
  all:      ["company"] as const,
  searches: () => [...companyKeys.all, "search"] as const,
  search:   (params) => [...companyKeys.searches(), params] as const,
  details:  () => [...companyKeys.all, "detail"] as const,
  detail:   (ico) => [...companyKeys.details(), ico] as const,
};
Why useQuery not useMutation? The search endpoint is a GET — it's idempotent and cacheable. TanStack Query automatically deduplicates identical in-flight requests and serves stale results while re-fetching. The query key includes all params so each unique search is cached separately.

class CompanySearchView(APIView):
    """GET /v1/companies/search/?legalForm=112&regionCode=19&..."""

    def get(self, request):
        # 1. Validate incoming query params (step 10)
        serializer = CompanySearchRequestSerializer(data=request.query_params)
        serializer.is_valid(raise_exception=True)

        # 2. Call the service layer (step 11)
        service = CompanyService()
        result = service.search(serializer.validated_data)

        # 3. Format the result back to JSON for the frontend (step 12)
        return Response(CompanySearchResultSerializer(result).data)
Thin controller pattern: The view doesn't contain any business logic. It follows a strict 3-step pattern: validate input → delegate to service → format output. Same pattern used by ARES and Justice views.

The heart of the hub. Builds a Django ORM query from validated params, filters the denormalized Company model, paginates, and returns results.
▼
backend/company/services.py 
class CompanyService:
    def search(self, params: dict) -> dict:
        qs = Company.objects.all()

        # Dynamic filter chain — only applied if param is present
        if ico  := params.get("ico"):    qs = qs.filter(ico=ico.zfill(8))
        if name := params.get("name"):   qs = qs.filter(name__icontains=name)
        if lf   := params.get("legalForm"):       qs = qs.filter(legal_form=lf)
        if rc   := params.get("regionCode"):      qs = qs.filter(region_code=rc)
        if ec   := params.get("employeeCategory"): qs = qs.filter(employee_category=ec)
        if rmin := params.get("revenueMin"):       qs = qs.filter(latest_revenue__gte=rmin)
        if rmax := params.get("revenueMax"):       qs = qs.filter(latest_revenue__lte=rmax)
        if nace := params.get("nace"):             qs = qs.filter(nace_primary=nace)

        # Status filter
        if params.get("status") == "active":   qs = qs.filter(is_active=True)
        elif params.get("status") == "inactive": qs = qs.filter(is_active=False)

        # Pagination & ordering
        total = qs.count()
        offset = params.get("offset", 0)
        limit  = params.get("limit", 25)
        companies = qs.order_by("-latest_revenue", "name")[offset:offset + limit]

        # Build response dict from model instances
        return {
            "totalCount": total,
            "offset": offset,
            "limit": limit,
            "companies": [
                {
                    "ico": c.ico,  "name": c.name,  "isActive": c.is_active,
                    "legalForm": c.legal_form,  "regionCode": c.region_code,
                    "regionName": c.region_name,  "employeeCategory": c.employee_category,
                    "latestRevenue": str(c.latest_revenue) if c.latest_revenue else None,
                    "nacePrimary": c.nace_primary,
                }
                for c in companies
            ],
        }
Why this is fast: The Company model has denormalized fields (copied from ARES/Justice during data ingestion). All filtering happens on indexed PostgreSQL columns — no external API calls, no Redis cache needed. Typical response: <50ms.

backend/company/models.py 
class Company(models.Model):
    """Hub entity — unique Czech company identified by ICO."""

    # Primary identifier
    ico        = CharField(max_length=8, unique=True, db_index=True)
    name       = CharField(max_length=500)
    is_active  = BooleanField(default=True)
    created_at = DateTimeField(auto_now_add=True)
    updated_at = DateTimeField(auto_now=True)

    # --- Denormalized search fields (populated from ARES/Justice) ---
    legal_form        = CharField(max_length=10,  db_index=True)
    region_code       = IntegerField(null=True,   db_index=True)
    region_name       = CharField(max_length=100)
    employee_category = CharField(max_length=50,  db_index=True)
    latest_revenue    = DecimalField(max_digits=15, db_index=True)
    nace_primary      = CharField(max_length=10,  db_index=True)

    class Meta:
        indexes = [
            # Composite index for common filter combo
            Index(fields=["legal_form", "region_code"], name="idx_company_form_region"),
        ]

    # Reverse FK relations (from spokes):
    # company.ares_records     → ares.EconomicSubject
    # company.justice_entities → justice.Entity
Denormalization explained: Instead of joining ARES + Justice tables at query time, fields like legal_form and region_code are copied into the Company model during data ingestion. This trades storage space for query speed — a single table scan with B-tree indexes is much faster than cross-table joins.

async getByIco(ico: string): Promise<CompanyDetail> {
  const normalizedIco = ico.padStart(8, "0");
  if (!/^\d{8}$/.test(normalizedIco))
    throw new CompanyApiError("Invalid ICO format.");

  const response = await apiClient.get<CompanyDetail>(
    `/companies/${normalizedIco}/`
  );
  return response.data;
}

            .order_by("-updated_at").first()
        justice_data = {
            "ico": justice_entity.ico,
            "name": justice_entity.name,
            "legalFormCode": justice_entity.legal_form_code,
            "legalFormName": justice_entity.legal_form_name,
            "courtName": justice_entity.court_name,
            "fileReference": justice_entity.file_reference,
            "registrationDate": ..., "deletionDate": ...,
            "isActive": justice_entity.is_active,
        } if justice_entity else None

        # 4. Fetch linked ARES record (reverse FK)
        ares_record = company.ares_records.first()

class AresService:
    def __init__(self):
        self.client = ares_client
        self.cache = CacheService(prefix="ares")
        self.outbound_throttle = GlobalOutboundThrottle(
            key="ares", max_requests=12, window=60  # 12 req/min
        )

    def get_by_ico(self, ico: str) -> dict:
        normalized = ico.zfill(8)

        # TIER 1: Hot cache (Redis, 1 hour TTL)
        cached = self.cache.get("detail", normalized)
        if cached is not None:
            return cached  # ← instant

        # TIER 2: Warm cache (PostgreSQL)
        db_record = EconomicSubject.objects.filter(
            ico=normalized
        ).first()
        if db_record and db_record.raw_data:
            result = parse_economic_subject(db_record.raw_data)
            self.cache.set(result, "detail", normalized)
            return result  # ← fast DB lookup

        # TIER 3: Cold — call external ARES API
        if not self.outbound_throttle.allow():
            raise ExternalAPIError("Rate limit", status_code=429)

        raw = self.client.get_by_ico(normalized)
        result = parse_economic_subject(raw)
        self.cache.set(result, "detail", normalized)
        self._persist_detail(normalized, result, raw)
        return result
3-tier cache explained: L1 Redis (1 hr) — fastest, in-memory. L2 PostgreSQL — the EconomicSubject model stores the entire ARES response as JSON. L3 ARES API — only called when both L1 and L2 miss. Rate-limited to 12 req/min. After an L3 hit, data is cached in both L1 and L2 for future requests.

class JusticeService:
    def __init__(self):
        self.cache = CacheService(prefix="justice", default_ttl=900)

    def get_entity_by_ico(self, ico: str) -> dict:
        normalized = ico.zfill(8)

        # 1. Check Redis cache (1 hour TTL)
        cached = self.cache.get("entity", normalized)
        if cached is not None:
            return cached

        # 2. Query PostgreSQL with eager loading
        entity = Entity.objects.filter(ico=normalized).first()
        facts = EntityFact.objects.filter(entity=entity) \
            .select_related("person") \
            .prefetch_related("addresses", "sub_facts") \
            .order_by("registration_date")

        # 3. Transform models to camelCase API dicts
        result = parse_entity_detail(entity, list(facts))

        # 4. Cache for 1 hour
        self.cache.set(result, "entity", normalized, ttl=3600)
        return result

    def get_entity_persons(self, ico: str) -> list[dict]:
        normalized = ico.zfill(8)
        entity = Entity.objects.filter(ico=normalized).first()

        facts_with_persons = EntityFact.objects.filter(
            entity=entity, person__isnull=False
        ).select_related("person").order_by("registration_date")

        return [parse_person_with_fact(f) for f in facts_with_persons]
